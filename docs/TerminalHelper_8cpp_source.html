29bool TerminalHelper::getTerminalSize(int& rows, int& cols) {
30 struct winsize w;
31 if (ioctl(STDOUT_FILENO, TIOCGWINSZ, &w) == 0) {
32 rows = w.ws_row;
33 cols = w.ws_col;
34 return true;
35 }
36 return false;
37}

39bool TerminalHelper::isTerminalSizeValid(int minRows, int minCols) {
40 int rows, cols;
41 if (getTerminalSize(rows, cols)) {
42 return rows >= minRows && cols >= minCols;
43 }
44 return false;
45}
