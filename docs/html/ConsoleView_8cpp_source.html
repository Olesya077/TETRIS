90void ConsoleView::ShowGhostFigure(Figure& figure, Field& field) {
91 if (!TerminalHelper::isTerminalSizeValid(24, 48)) {
92 return;
93 }
94
95 PictureField* pictureField = dynamic_cast<PictureField*>(&field);
96
97 int dropDepth = 0;
98 int currentX = figure.getstartx();
99 int currentY = figure.getstarty();
100 while (true) {
101 bool canMoveDown = true;
102 int x = figure.getstartx();
103 int y = figure.getstarty();
104
105 for (int i = 0; i < figure.getHeight() && canMoveDown; i++) {
106 for (int j = 0; j < figure.getWidth() && canMoveDown; j++) {
107 if (figure.getchar(i, j)) {
108 int fieldX = x + j;
109 int fieldY = y + i + dropDepth + 1;
110
111 if (!field.isValidPosition(fieldY, fieldX) || field.getch(fieldY, fieldX)) {
112 canMoveDown = false;
113 }
114 }
115 }
116 }
117
118 if (canMoveDown) {
119 dropDepth++;
120 } else {
121 break;
122 }
123 }
124
125 if (dropDepth > 0) {
126 int ghostX = currentX;
127 int ghostY = currentY + dropDepth;
128
129 for (int i = 0; i < figure.getHeight(); i++) {
130 for (int j = 0; j < figure.getWidth(); j++) {
131 if (figure.getchar(i, j)) {
132 int fieldX = ghostX + j;
133 int fieldY = ghostY + i;
134 bool isUnderCurrentFigure = false;
135 for (int fi = 0; fi < figure.getHeight() && !isUnderCurrentFigure; fi++) {
136 for (int fj = 0; fj < figure.getWidth() && !isUnderCurrentFigure; fj++) {
137 if (figure.getchar(fi, fj)) {
138 int currentFieldX = currentX + fj;
139 int currentFieldY = currentY + fi;
140 if (currentFieldX == fieldX && currentFieldY == fieldY) {
141 isUnderCurrentFigure = true;
142 }
143 }
144 }
145 }
146
147 if (fieldY >= 0 && fieldY < field.getHeight() &&
148 fieldX >= 0 && fieldX < field.getWidth() &&
149 !field.getch(fieldY, fieldX) &&
150 !isUnderCurrentFigure) {
151
152 TerminalHelper::moveCursorTo(fieldY, fieldX * 2);
153 std::cout << "\x1b[90m" << "▓▓" << "\x1b[0m";
154 }
155 }
156 }
157 }
158 }
