41void BucketField::initializeBucketShape() {
42 rowWidths.resize(fieldHeight);
43
44 int minWidth = 6;
45 int maxWidth = 22;
46
47 for (int row = 0; row < fieldHeight; row++) {
48 rowWidths[row] = minWidth + (maxWidth - minWidth) * row / (fieldHeight - 1);
49 }
50
51 fieldmatrix.clear();
52 fieldmatrix.resize(fieldHeight * fieldWidth, false);
53
54 for (int row = 0; row < fieldHeight; row++) {
55 int width = rowWidths[row];
56 int leftBound = (fieldWidth - width) / 2;
57 int rightBound = leftBound + width;
58
59 for (int col = 0; col < fieldWidth; col++) {
60 bool isWall = (col < leftBound) || (col >= rightBound);
61 setch(row, col, isWall);
62 }
63 }
64
65 for (int col = 0; col < fieldWidth; col++) {
66 setch(fieldHeight - 1, col, true);
67 }
68}

108int BucketField::clearFullLines() {
109 /**
110 * @brief Очищает линии с учетом формы ведра
111 * @return Количество очищенных линий
112 * @note Учитывает переменную ширину строк в ведре
113 */
114 int linesCleared = 0;
115
116 for (int row = fieldHeight - 2; row >= 1; row--) {
117 bool lineFull = true;
118
119 int width = getRowWidth(row);
120 int leftBound = (fieldWidth - width) / 2;
121 int rightBound = leftBound + width;
122
123 for (int col = leftBound; col < rightBound; col++) {
124 if (!getch(row, col)) {
125 lineFull = false;
126 break;
127 }
128 }
129
130 if (lineFull) {
131 linesCleared++;
132
133 for (int moveRow = row; moveRow >= 1; moveRow--) {
134 int moveWidth = getRowWidth(moveRow);
135 int moveLeftBound = (fieldWidth - moveWidth) / 2;
136 int moveRightBound = moveLeftBound + moveWidth;
137
138 int prevWidth = getRowWidth(moveRow - 1);
139 int prevLeftBound = (fieldWidth - prevWidth) / 2;
140 int prevRightBound = prevLeftBound + prevWidth;
141
142 for (int col = moveLeftBound; col < moveRightBound; col++) {
143 bool value = false;
144 std::string color = "";
145
146 if (moveRow > 0 && col >= prevLeftBound && col < prevRightBound) {
147 value = getch(moveRow - 1, col);
148 color = getColor(moveRow - 1, col);
149 }
150 setch(moveRow, col, value, color);
151 }
152 }
153
154 int topWidth = getRowWidth(0);
155 int topLeftBound = (fieldWidth - topWidth) / 2;
156 int topRightBound = topLeftBound + topWidth;
157 for (int col = topLeftBound; col < topRightBound; col++) {
158 setch(0, col, false);
159 }
160
161 row++;
162 }
163 }
164
165 return linesCleared;
166}

210int Field::clearFullLines() {
211 /**
212 * @brief Очищает полностью заполненные линии
213 * @return Количество очищенных линий
214 * @algorithm
215 * 1. Проверяем линии снизу вверх (кроме дна)
216 * 2. Если линия полная, сдвигаем все линии выше на одну вниз
217 * 3. Очищаем самую верхнюю линию
218 * 4. Повторяем для следующей линии (увеличиваем счетчик, так как линии сместились)
219 */
220 int linesCleared = 0;
221
222 for (int row = fieldHeight - 2; row >= 1; row--) {
223 bool lineFull = true;
224
225 for (int col = 1; col < fieldWidth - 1; col++) {
226 if (!getch(row, col)) {
227 lineFull = false;
228 break;
229 }
230 }
231
232 if (lineFull) {
233 linesCleared++;
234 for (int moveRow = row; moveRow >= 1; moveRow--) {
235 for (int col = 1; col < fieldWidth - 1; col++) {
236 bool value = getch(moveRow - 1, col);
237 std::string color = getColor(moveRow - 1, col);
238 setch(moveRow, col, value, color);
239 }
240 }
241 for (int col = 1; col < fieldWidth - 1; col++) {
242 setch(0, col, false);
243 }
244 row++;
245 }
246 }
247
248 return linesCleared;
249}
