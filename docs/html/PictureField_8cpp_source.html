106void PictureField::drawTriangle() {
107 for (int i = 0; i < 22 * 22; i++) {
108 targetPicture[i] = false;
109 }
110
111 int triangleHeight = 16;
112 int triangleBase = 19;
113
114 int floorRow = 20;
115 int startRow = floorRow - triangleHeight + 1;
116 int startCol = (22 - triangleBase) / 2;
117
118 for (int row = 0; row < triangleHeight; row++) {
119 int rowWidth = (row * triangleBase) / triangleHeight;
120
121 if (rowWidth < 1) rowWidth = 1;
122
123 int colStart = startCol + (triangleBase - rowWidth) / 2;
124 int colEnd = colStart + rowWidth;
125
126 for (int col = colStart; col < colEnd; col++) {
127 int fieldRow = startRow + row;
128 int fieldCol = col;
129
130 if (fieldRow >= 0 && fieldRow < 22 && fieldCol >= 0 && fieldCol < 22) {
131 targetPicture[fieldRow * 22 + fieldCol] = true;
132 }
133 }
134 }

152void PictureField::placeFigure(Figure& figure) {
153 int x = figure.getstartx();
154 int y = figure.getstarty();
155 bool placedInTarget = false;
156 bool touchedBorder = false;
157
158 for (int i = 0; i < figure.getHeight(); i++) {
159 for (int j = 0; j < figure.getWidth(); j++) {
160 if (figure.getchar(i, j)) {
161 int fieldx = x + j;
162 int fieldy = y + i;
163
164 if (isValidPosition(fieldy, fieldx)) {
165 if (isInTargetArea(fieldy, fieldx)) {
166 placedInTarget = true;
167 currentPicture[fieldy * 22 + fieldx] = true;
168 pictureColors[fieldy * 22 + fieldx] = figure.getcolor();
169 setch(fieldy, fieldx, true, figure.getcolor());
170 } else {
171 touchedBorder = true;
172 setch(fieldy, fieldx, true, figure.getcolor());
173 }
174 } else {
175 touchedBorder = true;
176 }
177 }
178 }
179 }
180
181 if (placedInTarget) {
182 if (isPictureComplete()) {
183 TerminalHelper::moveCursorTo(27, 0);
184 TerminalHelper::clearCurrentLine();
185 std::cout << "\x1b[32m" << "ПОЗДРАВЛЯЕМ! Вы собрали картинку: "
186 << getPictureName() << "!" << "\x1b[0m";
187 std::cout.flush();
188 usleep(3000000);
189 gameOver = true;
190 }
191 }
192
193 if (touchedBorder) {
194 TerminalHelper::moveCursorTo(27, 0);
195 TerminalHelper::clearCurrentLine();
196 std::cout << "\x1b[31m" << "ИГРА ОКОНЧЕНА! Часть фигуры вышла за границы картинки." << "\x1b[0m";
197 std::cout.flush();
198 usleep(3000000);
199 gameOver = true;
200 }
201}
