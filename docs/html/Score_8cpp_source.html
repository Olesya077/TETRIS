18void GameScore::loadScores() {
19 highScores.clear();

21
22 if (file.is_open()) {

50void GameScore::addScore(const std::string& name, int score) {
51 /**
52 * @brief Добавляет новый рекорд
53 * @param name Имя игрока
54 * @param score Количество очков
55 * @note Автоматически добавляет текущую дату
56 * Сортирует список и сохраняет только топ-10
57 */
58 std::time_t t = std::time(nullptr);
59 std::tm* now = std::localtime(&t);
60 std::stringstream ss;
61 ss << (now->tm_year + 1900) << "-"
62 << (now->tm_mon + 1) << "-"
63 << now->tm_mday;
64
65 HighScore newScore;
66 newScore.playerName = name;
67 newScore.score = score;
68 newScore.date = ss.str();
69
70 highScores.push_back(newScore);
71
72 std::sort(highScores.begin(), highScores.end(),
73 [](const HighScore& a, const HighScore& b) {
74 return a.score > b.score;
75 });
76
77 saveScores();
78}
