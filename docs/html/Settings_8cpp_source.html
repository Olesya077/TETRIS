92bool Settings::setControl(const std::string& action, char newKey) {
93 /**
94 * @brief Устанавливает новую клавишу для действия
95 * @param action Название действия
96 * @param newKey Новая клавиша
97 * @return true если клавиша успешно установлена
98 * @note Проверяет:
99 * 1. Существование действия
100 * 2. Валидность клавиши
101 * 3. Занятость клавиши другими действиями
102 * 4. Не является ли клавиша зарезервированной
103 */
104 newKey = std::tolower(newKey);
105
106 if (controls.find(action) == controls.end()) {
107 return false;
108 }
109
110 if (!isKeyAvailableForAction(action, newKey)) {
111 return false;
112 }
113
114 controls[action] = newKey;
115 saveSettings();
116 return true;
117}

172void Settings::saveSettings() {

189void Settings::loadSettings() {
190 std::ifstream file(settingsFile);
191 if (file.is_open()) {
192 std::string line;
193 std::string section;
194
195 while (std::getline(file, line)) {
196 if (line.empty() || line[0] == '#') continue;
197
198 if (line[0] == '[' && line[line.length()-1] == ']') {
199 section = line.substr(1, line.length()-2);
200 continue;
201 }
202
203 std::istringstream iss(line);
204 std::string key, value;
205 if (std::getline(iss, key, '=') && std::getline(iss, value)) {
206 if (section == "CONTROLS" && value.length() == 1) {
207 char c = value[0];
208 bool keyOccupied = false;
209 for (const auto& pair : controls) {
210 if (pair.first != key && pair.second == c) {
211 keyOccupied = true;
212 break;
213 }
214 }
215
216 if (!keyOccupied) {
217 controls[key] = c;
218 }
219 }
220 else if (section == "SETTINGS") {
221 try {
222 gameSettings[key] = std::stoi(value);
223 } catch (...) {
224 }
225 }
226 }
227 }
228 file.close();
229 }
230}
