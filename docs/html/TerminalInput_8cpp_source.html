12TerminalInput::TerminalInput() {
13 /**
14 * @brief Конструктор - настраивает терминал
15 * @note Отключает:
16 * - Канонический режим (ожидание Enter)
17 * - Эхо ввода
18 * Устанавливает:
19 * - Чтение по 1 символу
20 * - Без таймаута
21 */
22 struct termios term;
23 tcgetattr(STDIN_FILENO, &originalTerm);
24 term = originalTerm;
25 term.c_lflag &= ~(ICANON | ECHO);
26 term.c_cc[VMIN] = 1;
27 term.c_cc[VTIME] = 0;
28 tcsetattr(STDIN_FILENO, TCSANOW, &term);
29}

35char TerminalInput::getInput() {
36 /**
37 * @brief Читает один символ из терминала
38 * @return Прочитанный символ или 0 при ошибке
39 * @note Использует системный вызов read()
40 * Не блокирует выполнение если нет данных
41 */
42 char c = 0;
43 if (read(STDIN_FILENO, &c, 1) != 1) {
44 return 0;
45 }
46 return c;
47}
