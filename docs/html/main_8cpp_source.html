78 *
79 * @param argc Количество аргументов командной строки

87 * воспроизводить одинаковую последовательность фигур при разных запусках.
88 *
89 * Примеры использования:
90 * - @c ./tetris_game # Использует текущее время как seed
91 * - @c ./tetris_game 42 # Использует 42 как seed (фиксированная последовательность)
92 * - @c ./tetris_game 123456789 # Другой фиксированный seed
93 *
94 * @note Если аргумент не является числом, он будет проигнорирован и использовано
95 * текущее время в качестве seed.
96 */
97int main(int argc, char* argv[]) {
98 /**
99 * @section random_seed Инициализация генератора случайных чисел
100 *
101 * Генератор случайных чисел инициализируется одним из двух способов:
102 * 1. Если передан аргумент командной строки: используется как seed
103 * 2. Если аргументов нет: используется текущее время (time(NULL))
104 *
105 * Это обеспечивает либо воспроизводимость (с фиксированным seed), либо

107 */
108 if (argc > 1) {
109 int seed = std::atoi(argv[1]);
110 srand(seed);
111 std::cout << "Используется seed: " << seed << std::endl;
112 } else {
113 srand(time(NULL));
114 }
